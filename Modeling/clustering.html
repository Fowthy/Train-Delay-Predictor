def plot_dendrogram(model, **kwargs):
    # create the counts of samples under each node
    counts = np.zeros(model.children_.shape[0])
    n_samples = len(model.labels_)
    for i, merge in enumerate(model.children_):
        current_count = 0
        for child_idx in merge:
            if child_idx < n_samples:
                current_count += 1  # leaf node
            else:
                current_count += counts[child_idx - n_samples]
        counts[i] = current_count

    linkage_matrix = np.column_stack([model.children_, model.distances_,
                                      counts]).astype(float)

    # Plot the corresponding dendrogram
    dendrogram(linkage_matrix, **kwargs)

plt.figure(figsize=(13,14))

# Define labels for species
actual = df['isDelay'].unique()

# Define labels for clusters
clusters = df['Clustered isDelay'].unique()

# Sort cluster names in alphabetical order, i.e.,
# Cluster 0, Cluster 1, Cluster 2, etc.
clusters.sort()

# Determine the location for cluster labels 
x = np.arange(len(clusters))

# Define the width of the bars
width = 0.25

# Create the bar plot
fig, ax = plt.subplots(figsize=(16,10))
offset = -width
for spec in actual:
    nr_occurrences = []
    for clus in clusters:
        nr = df[(df['isDelay']==spec) & (df['Clustered isDelay']==clus)]['Clustered isDelay'].count()
        nr_occurrences.append(nr)
    rects = ax.bar(x + offset, nr_occurrences, width, label=spec)
    offset = offset + width

# Add text for labels, title and custom x-axis tick labels, etc.
ax.set_ylabel('Number of occurrences')
ax.set_title('Number of occurrences in cluster')
ax.set_xticks(x)
ax.set_xticklabels(clusters)
ax.legend()

fig.tight_layout()
plt.show()

Based on the plot we can see the two clusters and how the actual data was distributed between them. The 0 (blue bar) shows no delay and 1 (orange bar) shows delay. In the first cluster 0, we see that most of the train departures had no delay with count of above 30000, but there were also around 14000 departures with delays. We can assume that this is the cluster that will contain departures without delays. The data in the second cluster is almost balanced - there were around 20000 departures without delays and ~13000 with delays.
